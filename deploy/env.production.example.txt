# Copy to the repo root as `.env.production.local` on the VPS and fill in.
# The root `.env*.local` is gitignored. Values mirror the existing Vercel

# KV / Turso backends.

#   cp deploy/env.production.example.txt .env.production.local
#   $EDITOR .env.production.local

# --- Vercel KV (reused as-is) ----------------------------------------------
# Used by weather/POI/quake/warning proxies (cache + rate-limit) and by
# Web Push subscription storage. Without these the app still boots but
# rate-limit/cache fall back to a process-local LRU and Push stops working.
KV_URL=
KV_REST_API_URL=
KV_REST_API_TOKEN=
KV_REST_API_READ_ONLY_TOKEN=

# --- Datasets DB (Turso libsql) --------------------------------------------
# Optional. If unset the app uses the build-time data/datasets.sqlite file,
# which is frozen at build time. Point at Turso to keep bus/events fresh via
# the existing hourly GitHub Actions sync. The "required in production" guard
# only fires under VERCEL_ENV=production, so leaving these blank is fine here.
DATASETS_DB_URL=
DATASETS_DB_AUTH_TOKEN=

# --- Site / routing --------------------------------------------------------
# basePath for the Tailscale Funnel deployment. Keep in sync with the
# `tailscale funnel --set-path` value in deploy.sh.
NEXT_PUBLIC_BASE_PATH=/my-koto
# Public origin incl. basePath. Used for OG images and CORS. Set this to the
# Funnel hostname after `tailscale funnel status`, then rebuild.
NEXT_PUBLIC_SITE_URL=https://your-host.your-tailnet.ts.net/my-koto

# --- Web Push (durable subscriptions live in KV above) ---------------------
NEXT_PUBLIC_VAPID_PUBLIC_KEY=
VAPID_PRIVATE_KEY=
VAPID_SUBJECT=mailto:you@example.com
# Bearer token guarding /api/push/dispatch. Must match the value used by the
# dispatch timer (push-dispatch.timer / curl).
PUSH_DISPATCH_SECRET=

# --- Optional --------------------------------------------------------------
# Discord webhook for KV-failure / data-fetch failure notifications.
DISCORD_WEBHOOK=
# Listen port for `next start`. deploy.sh / the systemd unit default to 3000.
PORT=3000
